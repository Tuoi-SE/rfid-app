```typescript
/** Marker for auto-generated temp passwords — triggers force-password-change on login */
private static readonly TEMP_PASSWORD_MARKER = '__TEMP_PASSWORD__';

/**
 * Send password reset email.
 * Throws AUTH_EMAIL_NOT_FOUND (404) if email not registered.
 * Uses throttle-safe: always returns success to prevent email enumeration.
 */
async forgotPassword(email: string): Promise<void> {
  const user = await this.prisma.user.findFirst({
    where: { email, deletedAt: null },
  });
  // Always return success — do NOT reveal whether email exists (prevents enumeration)
  if (!user) return;

  // Invalidate any existing reset tokens for this user
  await this.prisma.passwordResetToken.updateMany({
    where: { userId: user.id, usedAt: null },
    data: { usedAt: new Date() },
  });

  // Generate and hash token
  const rawToken = crypto.randomBytes(32).toString('base64url');
  const hashedToken = crypto.createHash('sha256').update(rawToken).digest('hex');
  const expiresAt = new Date(Date.now() + 15 * 60 * 1000); // 15 minutes

  await this.prisma.passwordResetToken.create({
    data: { token: hashedToken, userId: user.id, expiresAt },
  });

  const baseUrl = this.config.get('APP_BASE_URL', 'http://localhost:3000');
  const resetUrl = `${baseUrl}/reset-password?token=${rawToken}`;

  await this.emailService.sendPasswordResetEmail(user.email!, resetUrl);
}

/**
 * Reset password using a valid token.
 * Throws AUTH_INVALID_RESET_TOKEN (400) if token not found, expired, or already used.
 */
async resetPassword(token: string, newPassword: string): Promise<void> {
  const hashedToken = crypto.createHash('sha256').update(token).digest('hex');

  const resetToken = await this.prisma.passwordResetToken.findFirst({
    where: {
      token: hashedToken,
      usedAt: null,
      expiresAt: { gt: new Date() },
    },
  });

  if (!resetToken) {
    throw new BusinessException(
      'Liên kết khôi phục không hợp lệ hoặc đã hết hạn',
      AUTH_ERROR_CODES.INVALID_RESET_TOKEN,
      HttpStatus.BAD_REQUEST,
    );
  }

  const hashedPassword = await bcrypt.hash(newPassword, SALT_ROUNDS);

  await this.prisma.$transaction([
    this.prisma.user.update({
      where: { id: resetToken.userId },
      data: { password: hashedPassword },
    }),
    this.prisma.passwordResetToken.update({
      where: { id: resetToken.id },
      data: { usedAt: new Date() },
    }),
  ]);
}

/**
 * Register a new user and send email verification.
 * Throws AUTH_EMAIL_ALREADY_EXISTS (409) if email already taken.
 */
async register(email: string, username: string, role?: string): Promise<{ id: string; email: string; username: string; role: string }> {
  // Check email uniqueness
  const existingEmail = await this.prisma.user.findFirst({ where: { email, deletedAt: null } });
  if (existingEmail) {
    throw new BusinessException(
      'Email đã được sử dụng bởi tài khoản khác',
      AUTH_ERROR_CODES.EMAIL_ALREADY_EXISTS,
      HttpStatus.CONFLICT,
    );
  }

  // Check username uniqueness
  const existingUsername = await this.prisma.user.findFirst({ where: { username, deletedAt: null } });
  if (existingUsername) {
    throw new BusinessException(
      'Tên đăng nhập đã được sử dụng',
      'USER_USERNAME_EXISTS',
      HttpStatus.CONFLICT,
    );
  }

  // Generate temp password (marked as auto-generated)
  const tempPassword = crypto.randomBytes(16).toString('base64');
  const hashedPassword = await bcrypt.hash(tempPassword, SALT_ROUNDS);

  const user = await this.prisma.user.create({
    data: {
      email,
      username,
      password: hashedPassword, // Stored as temp, requiresPasswordChange = true
      role: (role as any) || 'WAREHOUSE_MANAGER',
      isEmailVerified: false,
    },
    select: { id: true, email: true, username: true, role: true },
  });

  // Generate and store verification token
  const rawToken = crypto.randomBytes(32).toString('base64url');
  const hashedToken = crypto.createHash('sha256').update(rawToken).digest('hex');
  const expiresAt = new Date(Date.now() + 24 * 60 * 60 * 1000); // 24 hours

  await this.prisma.emailVerificationToken.create({
    data: { token: hashedToken, userId: user.id, expiresAt },
  });

  const baseUrl = this.config.get('APP_BASE_URL', 'http://localhost:3000');
  const verifyUrl = `${baseUrl}/verify-email?token=${rawToken}`;

  await this.emailService.sendVerificationEmail(email, verifyUrl);

  return user;
}

/**
 * Verify email with a valid token.
 * After verification, user can log in normally.
 * Throws AUTH_INVALID_VERIFICATION_TOKEN (400) if token invalid/expired.
 */
async verifyEmail(token: string): Promise<void> {
  const hashedToken = crypto.createHash('sha256').update(token).digest('hex');

  const verificationToken = await this.prisma.emailVerificationToken.findFirst({
    where: {
      token: hashedToken,
      usedAt: null,
      expiresAt: { gt: new Date() },
    },
  });

  if (!verificationToken) {
    throw new BusinessException(
      'Liên kết xác thực không hợp lệ hoặc đã hết hạn',
      AUTH_ERROR_CODES.INVALID_VERIFICATION_TOKEN,
      HttpStatus.BAD_REQUEST,
    );
  }

  await this.prisma.$transaction([
    this.prisma.user.update({
      where: { id: verificationToken.userId },
      data: { isEmailVerified: true },
    }),
    this.prisma.emailVerificationToken.update({
      where: { id: verificationToken.id },
      data: { usedAt: new Date() },
    }),
  ]);
}

/**
 * Change password for authenticated user (after forced reset).
 * Validates current password before allowing change.
 */
async changePassword(userId: string, currentPassword: string, newPassword: string): Promise<void> {
  const user = await this.prisma.user.findUnique({ where: { id: userId } });
  if (!user) {
    throw new BusinessException('Không tìm thấy tài khoản', 'USER_NOT_FOUND', HttpStatus.NOT_FOUND);
  }

  const isMatch = await bcrypt.compare(currentPassword, user.password);
  if (!isMatch) {
    throw new BusinessException(
      'Mật khẩu hiện tại không đúng',
      'AUTH_INVALID_PASSWORD',
      HttpStatus.UNAUTHORIZED,
    );
  }

  const hashedPassword = await bcrypt.hash(newPassword, SALT_ROUNDS);
  await this.prisma.user.update({
    where: { id: userId },
    data: { password: hashedPassword },
  });
}
```

/**

 * @body { email, username, role? }
 * @returns { id, email, username, role }
 * @error AUTH_EMAIL_ALREADY_EXISTS (409), USER_USERNAME_EXISTS (409)

@Throttle({ default: { limit: 5, ttl: 60000 } })
@Public()
@Post('register')
@ResponseMessageDecorator.withMessage('Đăng ký thành công. Vui lòng kiểm tra email để xác thực tài khoản.')

/**

 * @query token
 * @returns 200
 * @error AUTH_INVALID_VERIFICATION_TOKEN (400)

@Get('verify-email')
@ResponseMessageDecorator.withMessage('Xác thực email thành công. Bạn có thể đăng nhập ngay.')

  await this.authService.verifyEmail(token);
  return null;

  @Body() dto: ChangePasswordDto,
