```prisma
enum PasswordResetTokenType {
  RESET   // For forgot password flow (6-digit numeric)
  WELCOME // For welcome email on user creation (generated by SuperAdmin)
}

model PasswordResetToken {
  id        String                @id @default(uuid())
  token     String                @unique           // SHA-256 hash of token
  userId    String
  user      User                  @relation(fields: [userId], references: [id], onDelete: Cascade)
  type      PasswordResetTokenType @default(RESET)
  expiresAt DateTime
  usedAt    DateTime?
  createdAt DateTime              @default(now())

  @@index([token])
  @@index([userId])
  @@index([type])
}
```

```typescript
import { Injectable } from '@nestjs/common';
import * as nodemailer from 'nodemailer';
import { ConfigService } from '@nestjs/config';

@Injectable()
export class EmailService {
  private transporter: nodemailer.Transporter;

  constructor(private config: ConfigService) {
    this.transporter = nodemailer.createTransport({
      host: this.config.getOrThrow<string>('SMTP_HOST'),
      port: Number(this.config.get<number>('SMTP_PORT', 587)),
      secure: Number(this.config.get<number>('SMTP_PORT', 587)) === 465,
      auth: {
        user: this.config.getOrThrow<string>('SMTP_USER'),
        pass: this.config.getOrThrow<string>('SMTP_PASS'),
      },
    });
  }

  /**
   * Send welcome email when SuperAdmin creates a new user.
   * Contains the auto-generated temporary password.
   */
  async sendWelcomeEmail(
    to: string,
    username: string,
    tempPassword: string,
    loginUrl: string,
  ): Promise<void> {
    const html = this.buildWelcomeEmailHtml(username, tempPassword, loginUrl);
    await this.transporter.sendMail({
      from: this.config.get('SMTP_FROM', 'noreply@rfidinventory.com'),
      to,
      subject: 'Chào mừng bạn — RFID Inventory',
      html,
    });
  }

  /**
   * Send password reset email with 6-digit numeric token.
   */
  async sendPasswordResetEmail(
    to: string,
    resetCode: string,
    username: string,
    resetUrl: string,
  ): Promise<void> {
    const html = this.buildPasswordResetHtml(resetCode, username, resetUrl);
    await this.transporter.sendMail({
      from: this.config.get('SMTP_FROM', 'noreply@rfidinventory.com'),
      to,
      subject: 'Khôi phục mật khẩu — RFID Inventory',
      html,
    });
  }
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| client → API | Untrusted input: email, password, token values |
| SMTP server → Email | Email delivery; links are guessable tokens |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|----------------|
| T-14-01 | Tampering | PasswordResetToken token field | mitigate | Store SHA-256 hash of token, not raw token (matching RefreshToken pattern) |
| T-14-02 | Tampering | EmailVerificationToken token field | mitigate | Store SHA-256 hash of token |
| T-14-03 | Information Disclosure | Email address exposure | accept | Email addresses are not considered highly sensitive in this B2B context |
| T-14-04 | Denial of Service | Forgot password abuse | mitigate | Existing throttle: 10/min on auth endpoints; rate-limit forgot-password separately |
| T-14-05 | Repudiation | Token reuse | mitigate | Mark token as `usedAt` after single use; reject reused tokens |
| T-14-06 | Spoofing | SMTP sender | mitigate | SMTP_FROM env var controlled, not user input |
</threat_model>
