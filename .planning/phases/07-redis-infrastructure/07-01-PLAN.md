          port: configService.get<number>('REDIS_PORT', 6379),
          ttl: 60 * 1000, // 60 seconds default TTL (per D-06)

    }),

  <name>Task 5: Update /health endpoint with Redis status</name>
  <files>backend/src/health/health.controller.ts</files>
  <read_first>

  </read_first>
  <action>
    Update HealthController to include Redis connection status per D-09 (user decision).

    Inject CacheModule and add Redis status to the health check response:

    ```typescript
    import { Cache } from '@nestjs/cache-manager';
    import { ConfigService } from '@nestjs/config';

    @Controller('api/health')
    export class HealthController {
      constructor(
        private prisma: PrismaService,
        private cacheManager: Cache,
        private configService: ConfigService,
      ) {}

      @Get()
      @ResponseMessage('Kiểm tra sức khỏe hệ thống thành công')
      async check() {
        const redisHost = this.configService.get<string>('REDIS_HOST', 'localhost');
        const redisPort = this.configService.get<number>('REDIS_PORT', 6379);

        let redisStatus = 'degraded';
        try {
          await this.cacheManager.get('__health_check__');
          redisStatus = 'healthy';
        } catch {
          redisStatus = 'degraded';
        }

        try {
          await this.prisma.$queryRaw`SELECT 1`;
          return {
            status: 'ok',
            database: 'connected',
            redis: redisStatus,
            redisHost,
            redisPort,
            timestamp: new Date().toISOString()
          };
        } catch {
          return {
            status: 'error',
            database: 'disconnected',
            redis: redisStatus,
            redisHost,
            redisPort,
            timestamp: new Date().toISOString()
          };
        }
      }
    }
    ```
  </action>
